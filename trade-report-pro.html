async function syncIB() { const btn = document.getElementById('syncBtn'); const resDiv = document.getElementById('syncResult'); const statusText = document.getElementById('syncStatusText'); // שליפת הגדרות לפי המצב הנוכחי (Live/Demo) const settings = currentAccountMode === 'live' ? userSettings.ibLive : userSettings.ibDemo; if (!settings.token || !settings.queryId) { alert('נא להגדיר Token ו-Query ID בהגדרות'); return; } btn.disabled = true; btn.innerHTML = ' מסנכרן...'; resDiv.style.display = 'none'; // שימוש ב-Proxy שבנית ב-Cloudflare const proxy = "https://long-voice-979fib-proxy.israelco7890.workers.dev/?url="; const ibUrl = `https://www.interactivebrokers.com/Universal/servlet/FlexStatementService.SendRequest?t=${settings.token}&q=${settings.queryId}&v=3`; try { const response = await fetch(proxy + encodeURIComponent(ibUrl)); const text = await response.text(); const parser = new DOMParser(); const xml = parser.parseFromString(text, "text/xml"); // בדיקה אם אינטראקטיב החזירה שגיאה const errorCode = xml.querySelector('errorCode'); if (errorCode) { throw new Error("שגיאת IB: " + xml.querySelector('errorMessage').textContent); } // חיפוש עסקאות בפורמט של Activity Flex Query (שדה Trades/Execution) const executions = xml.querySelectorAll('Trade'); // זה השדה הנכון ב-Activity Query let count = 0; let newTrades = []; executions.forEach(ex => { // שליפת הנתונים הקריטיים מה-XML const symbol = ex.getAttribute('symbol'); const side = ex.getAttribute('side'); // BUY/SELL const qty = Math.abs(parseFloat(ex.getAttribute('quantity'))); const price = parseFloat(ex.getAttribute('tradePrice')); const date = ex.getAttribute('dateTime'); const pnl = parseFloat(ex.getAttribute('realizedPnl') || 0); // כאן מגיע הרווח! const comm = parseFloat(ex.getAttribute('ibCommission') || 0); if (symbol) { newTrades.push({ id: 'ib-' + ex.getAttribute('tradeID') || Math.random(), symbol: symbol, type: side === 'BUY' ? 'Long' : 'Short', entryPrice: price, exitPrice: pnl !== 0 ? (price + (pnl/qty)) : 0, // חישוב מחיר יציאה משוער אם חסר quantity: qty, status: pnl !== 0 ? 'סגורה' : 'פתוחה', pnl: pnl + comm, // רווח נקי אחרי עמלה date: date, notes: 'סונכרן מ-IB' }); count++; } }); if (count > 0) { // שמירה למסד הנתונים (Firebase) allTrades = [...newTrades, ...allTrades.filter(t => !t.id.startsWith('ib-'))]; await saveTrades(); renderDashboard(); renderTradesTable(); resDiv.className = 'sync-result success'; resDiv.innerText = `הסנכרון הושלם! נוספו/עודכנו ${count} עסקאות.`; } else { resDiv.className = 'sync-result error'; resDiv.innerText = 'לא נמצאו עסקאות חדשות בטווח התאריכים המוגדר.'; } } catch (err) { console.error(err); resDiv.className = 'sync-result error'; resDiv.innerText = 'שגיאה: ' + err.message; } finally { resDiv.style.display = 'block'; btn.disabled = false; btn.innerText = '⟳ סנכרן מ-IB'; } }
